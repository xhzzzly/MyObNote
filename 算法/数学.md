## 欧拉筛

```c++
// 欧拉筛 + 最小素因数表
vector<int> primes;  // 素数数组
vector<bool> pvis;   // 标记数组，false为素数
vector<int> minp;    // 每个数的最小素因数
void euler_sieve() {

    primes.clear();
    pvis.assign(n + 3, false);
    minp.resize(n + 3);
    for (long long i = 2; i <= n; ++i) {
        if (!pvis[i]) {
            primes.emplace_back(i);
            minp[i] = i;
        }
        for (int j = 0; j < primes.size(); ++j) {
            long long num = i * primes[j];   // long long 防爆
            if (num > n)
                break;
            pvis[num] = true;
            minp[num] = primes[j];
            if (i % primes[j] == 0)
                break;
        }
    }
}
```
